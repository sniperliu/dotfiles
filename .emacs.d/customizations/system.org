  :config
  (setq user-full-name "Liu Hao")
  (setq user-mail-address "sniperliuhao@gmail.com")

  ;; locales to UTF8
  (set-language-environment 'utf-8)

  (set-selection-coding-system 'utf-8)
  (prefer-coding-system 'utf-8))
